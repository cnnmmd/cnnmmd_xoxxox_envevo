FROM ubuntu:24.04

ENV OPTAPT=""
#ENV OPTAPT="-o Acquire::AllowInsecureRepositories=true -o Acquire::AllowDowngradeToInsecureRepositories=true"
RUN \
  apt-get ${OPTAPT} -y update && \
  apt-get ${OPTAPT} -y upgrade && \
  apt-get ${OPTAPT} -y install --no-install-recommends xauth && \
  apt-get ${OPTAPT} -y install --no-install-recommends openssh-server && \
  apt-get ${OPTAPT} -y install --no-install-recommends python3 python3-venv && \
  apt-get ${OPTAPT} -y install --no-install-recommends software-properties-common

RUN \
  add-apt-repository ppa:deadsnakes/ppa && \
  apt-get ${OPTAPT} -y update && \
  apt-get ${OPTAPT} -y install --no-install-recommends python3.10 python3.10-venv python3.10-dev

RUN python3.10 -m venv /env/python

RUN \
  apt-get ${OPTAPT} -y install --no-install-recommends build-essential && \
  apt-get ${OPTAPT} -y install --no-install-recommends cmake && \
  apt-get ${OPTAPT} -y install --no-install-recommends xorg-dev && \
  apt-get ${OPTAPT} -y install --no-install-recommends libglu1-mesa-dev && \
  /env/python/bin/pip install evogym --upgrade

RUN \
  apt-get ${OPTAPT} -y install --no-install-recommends python3.10-tk && \
  /env/python/bin/pip install glfw && \
  /env/python/bin/pip install PyOpenGL && \
  /env/python/bin/pip install ttkbootstrap && \
  rm -rf /var/lib/apt/lists/*

RUN mkdir -p /run/sshd

ENV USRAPP=usrapp
RUN useradd -m -s /bin/bash ${USRAPP}
RUN passwd -l ${USRAPP}

ENV DIRSSH=/home/${USRAPP}/.ssh
ENV ATHSSH=authorized_keys
RUN mkdir -p ${DIRSSH} && chmod 700 ${DIRSSH}
COPY --chmod=600 ${ATHSSH} ${DIRSSH}
RUN chown -R ${USRAPP}:${USRAPP} ${DIRSSH}

ENV DIRPYT=/env/python
RUN python3 -m venv ${DIRPYT}
RUN ${DIRPYT}/bin/pip3 install --no-cache-dir gymnasium[classic-control]
